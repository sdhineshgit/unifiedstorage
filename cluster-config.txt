# Set hacluster password on ALL nodes
sudo passwd hacluster

# Re-authenticate ALL nodes
sudo pcs cluster auth -u hacluster -p [your-password]

sudo pcs cluster start --enable

# Restart pcsd
sudo systemctl restart pcsd

#Cluster status
sudo pcs status
